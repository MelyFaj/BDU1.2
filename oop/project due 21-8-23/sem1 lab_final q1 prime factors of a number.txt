/*
An integer is entered through the keyboard. Write a function to obtain the prime factors of this number. For example, prime factors of 24 are 2, 2, 2 and 3 whereas prime factors of 35 are 5 and 7
*/

#include<stdio.h>
int prm(int n){
int n;
    scanf("%d", &n);
    for(int i=2; i<n; i++){
        if(n%i==0)
            return 0;
    }
    return 1;
}
int prm_fact(int n){
    int a;
    for(int i=2; i<=n; i++){
            a=prm(i);
            if(n%a==0 )
                printf("%d\n", n/a);
    }
}
int main(){
    
    prm_fact(n);
    return 0;
}
